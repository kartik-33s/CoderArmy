08 int binarySearch(vector<int>& arr, int target) {
09     int left = 0, right = arr.size() - 1;
10     
11     while (left <= right) {
12         int mid = left + (right - left) / 2;
13         
14         if (arr[mid] == target) {
15             return mid; // Found
16         }
17         else if (arr[mid] < target) {
18             left = mid + 1;
19         } else {
20             right = mid - 1;
21         }
22     }
23     return -1; // Not found
24 }
